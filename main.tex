  研究の結果，\LaTeX でクリスマスカードが作成できることが分かった．
\end{abstract}

\section{Christmas Tree}

\begin{figure}[ht]
  \begin{center}
    \color{green}
    \begin{tikzpicture}[level distance = 1cm,
        level 1/.style={sibling distance=40mm},
        level 2/.style={sibling distance=20mm},
        level 3/.style={sibling distance=10mm}
      ]
      \node[yellow] {\Large ★}
      child {node[red] {●}
          child {node [yellow] {●}
              child {node [red] {●}}
              child {node [purple] {●}}
            }
          child {node [blue] {●}
              child {node [orange] {●}}
              child {node [white] {●}}}
        }
      child {node [orange] {●}
          child {node [white] {●}
              child {node [blue] {●}}
              child {node [red] {●}}}
          child {node [red] {●}
              child {node [purple] {●}}
              child {node [yellow] {●}}}
        };
    \end{tikzpicture}
  \end{center}
  \caption{Ti\textit{k}Z で描いた Christmas Tree}
  \label{fig:image}
\end{figure}
